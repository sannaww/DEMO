using System;
using System.Drawing;
using System.IO;
using System.Windows.Forms;
using Npgsql;

namespace ShoeStoreApp
{
    public partial class ProductEditForm : Form
    {
        string cs = "Host=localhost;Port=5432;Database=DEMO;Username=postgres;Password=123123";
        int productId = 0; // 0 - добавление, иначе редактирование
        string photo = "Res/picture.png";
        public ProductEditForm()
        {
            InitializeComponent();
            Text = "Добавление товара";
            BackColor = Color.White;
        }
        public ProductEditForm(int id) : this()
        {
            productId = id;
            Text = "Редактирование товара";
        }
        private void ProductEditForm_Load(object sender, EventArgs e)
        {
            // Загружаем списки
            LoadCombo(comboBoxName, "product_names");
            LoadCombo(comboBoxCategory, "categories");
            LoadCombo(comboBoxProducer, "producers");
            LoadCombo(comboBoxProvider, "providers");
            if (productId != 0) // Если редактирование — заполняем поля
                LoadProduct();
            ShowPhoto();
        }
        void LoadCombo(ComboBox box, string table)
        {
            // Заполняем ComboBox обычным текстом
            box.Items.Clear();
            using (var con = new NpgsqlConnection(cs))
            {
                con.Open();
                var cmd = new NpgsqlCommand("SELECT name FROM " + table + " ORDER BY name", con);
                var r = cmd.ExecuteReader();
                while (r.Read())
                    box.Items.Add(r["name"].ToString());
            }
        }
        int GetId(string table, string name)
        {
            // Получаем id выбранного значения
            using (var con = new NpgsqlConnection(cs))
            {
                con.Open();
                var cmd = new NpgsqlCommand("SELECT id FROM " + table + " WHERE name=@name", con);
                cmd.Parameters.AddWithValue("@name", name);
                return Convert.ToInt32(cmd.ExecuteScalar());
            }
        }
        void LoadProduct()
        {
            // Загружаем товар для редактирования
            using (var con = new NpgsqlConnection(cs))
            {
                con.Open();
                string sql = @"
                    SELECT p.*, pn.name AS pname, c.name AS cname, 
                           pr.name AS prname, pv.name AS pvname
                    FROM products p
                    JOIN product_names pn ON p.product_name_id = pn.id
                    JOIN categories c ON p.category_id = c.id
                    JOIN producers pr ON p.producer_id = pr.id
                    JOIN providers pv ON p.provider_id = pv.id
                    WHERE p.id=@id";
                var cmd = new NpgsqlCommand(sql, con);
                cmd.Parameters.AddWithValue("@id", productId);
                var r = cmd.ExecuteReader();
                if (r.Read())
                {
                    textBoxArticle.Text = r["article"].ToString();
                    comboBoxName.Text = r["pname"].ToString();
                    comboBoxCategory.Text = r["cname"].ToString();
                    comboBoxProducer.Text = r["prname"].ToString();
                    comboBoxProvider.Text = r["pvname"].ToString();
                    textBoxPrice.Text = r["price"].ToString();
                    textBoxUnit.Text = r["unit"].ToString();
                    textBoxCount.Text = r["amount_in_stock"].ToString();
                    textBoxDiscount.Text = r["discount"].ToString();
                    textBoxDescription.Text = r["description"].ToString();
                    photo = r["photo"].ToString();
                }
            }
        }
        void ShowPhoto()
        {
            // Показываем фото или заглушку
            string path = Path.Combine(Application.StartupPath, photo.Replace("/", "\\"));
            if (!File.Exists(path))
                path = Path.Combine(Application.StartupPath, "Res\\picture.png");
            pictureBoxProduct.Image = Image.FromFile(path);
            pictureBoxProduct.SizeMode = PictureBoxSizeMode.Zoom;
        }
        private void buttonSave_Click_1(object sender, EventArgs e)
        {
            decimal price, count, discount;
            // Проверка заполнения
            if (textBoxArticle.Text == "" || comboBoxName.Text == "" || comboBoxCategory.Text == "" ||
                comboBoxProducer.Text == "" || comboBoxProvider.Text == "" ||
                textBoxPrice.Text == "" || textBoxCount.Text == "")
            {
                MessageBox.Show("Заполните обязательные поля");
                return;
            }
            // Проверка чисел
            if (!decimal.TryParse(textBoxPrice.Text, out price) ||
                !decimal.TryParse(textBoxCount.Text, out count) ||
                !decimal.TryParse(textBoxDiscount.Text, out discount))
            {
                MessageBox.Show("Цена, количество и скидка должны быть числами");
                return;
            }
            using (var con = new NpgsqlConnection(cs))
            {
                con.Open();
                string sql = productId == 0 // Добавление или редактирование
                    ? @"INSERT INTO products
                       (article, product_name_id, category_id, producer_id, provider_id,
                        price, unit, amount_in_stock, discount, description, photo)
                       VALUES
                       (@a,@n,@c,@pr,@pv,@price,@unit,@count,@disc,@desc,@photo)"
                    : @"UPDATE products SET
                       article=@a, product_name_id=@n, category_id=@c, producer_id=@pr,
                       provider_id=@pv, price=@price, unit=@unit, amount_in_stock=@count,
                       discount=@disc, description=@desc, photo=@photo
                       WHERE id=@id";
                var cmd = new NpgsqlCommand(sql, con);
                cmd.Parameters.AddWithValue("@a", textBoxArticle.Text);
                cmd.Parameters.AddWithValue("@n", GetId("product_names", comboBoxName.Text));
                cmd.Parameters.AddWithValue("@c", GetId("categories", comboBoxCategory.Text));
                cmd.Parameters.AddWithValue("@pr", GetId("producers", comboBoxProducer.Text));
                cmd.Parameters.AddWithValue("@pv", GetId("providers", comboBoxProvider.Text));
                cmd.Parameters.AddWithValue("@price", price);
                cmd.Parameters.AddWithValue("@unit", textBoxUnit.Text);
                cmd.Parameters.AddWithValue("@count", count);
                cmd.Parameters.AddWithValue("@disc", discount);
                cmd.Parameters.AddWithValue("@desc", textBoxDescription.Text);
                cmd.Parameters.AddWithValue("@photo", photo);
                if (productId != 0)
                    cmd.Parameters.AddWithValue("@id", productId);
                cmd.ExecuteNonQuery();
            }
            MessageBox.Show("Данные сохранены");
            Close();
        }
        private void buttonPhoto_Click_1(object sender, EventArgs e)
        {
            OpenFileDialog d = new OpenFileDialog();
            d.Filter = "Картинки|*.jpg;*.jpeg;*.png";

            if (d.ShowDialog() == DialogResult.OK)
            {
                string file = Path.GetFileName(d.FileName);
                string folder = Path.Combine(Application.StartupPath, "Res");

                if (!Directory.Exists(folder))
                    Directory.CreateDirectory(folder);

                File.Copy(d.FileName, Path.Combine(folder, file), true);

                photo = "Res/" + file;
                ShowPhoto();
            }
        }
        private void buttonBack_Click_1(object sender, EventArgs e)
        {
            Close();
        }
    }
}
