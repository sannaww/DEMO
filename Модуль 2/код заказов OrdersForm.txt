using System;
using System.Data;
using System.Drawing;
using System.Windows.Forms;
using Npgsql;

namespace ShoeStoreApp
{
    public partial class OrdersForm : Form
    {
        string cs = "Host=localhost;Port=5432;Database=DEMO;Username=postgres;Password=123123";
        string userName, userRole;

        public OrdersForm(string name, string role)
        {
            InitializeComponent();
            userName = name;
            userRole = role;
            BackColor = Color.White;
            Load += OrdersForm_Load;
        }

        private void OrdersForm_Load(object sender, EventArgs e)
        {
            labelUser.Text = "Пользователь: " + userName + " (" + userRole + ")";
            buttonAddOrder.Visible = userRole == "Администратор";
            LoadOrders();
        }

        void LoadOrders()
        {
            // Загрузка заказов
            string sql = @"
            SELECT o.id, o.creation_date, o.delivery_date, s.name AS status,
                   pp.city || ', ' || pp.street || ', ' || COALESCE(pp.building, '') AS point,
                   STRING_AGG(p.article, ', ') AS articles
            FROM orders o
            JOIN order_statuses s ON o.status_id = s.id
            JOIN pickup_points pp ON o.pickup_point_id = pp.id
            LEFT JOIN product_in_order po ON po.order_id = o.id
            LEFT JOIN products p ON po.product_id = p.id
            GROUP BY o.id, s.name, pp.city, pp.street, pp.building
            ORDER BY o.id";

            DataTable dt = new DataTable();

            using (var con = new NpgsqlConnection(cs))
                new NpgsqlDataAdapter(sql, con).Fill(dt);

            flowOrders.Controls.Clear();

            foreach (DataRow row in dt.Rows)
                AddCard(row);
        }

        void AddCard(DataRow row)
        {
            int id = Convert.ToInt32(row["id"]);

            Panel card = new Panel();
            card.Size = new Size(980, 210);
            card.Margin = new Padding(10);
            card.BorderStyle = BorderStyle.FixedSingle;
            card.BackColor = Color.White;

            Label info = new Label();
            info.Bounds = new Rectangle(20, 15, 620, 140);
            info.Font = new Font("Times New Roman", 12);
            info.Text =
                "Артикул заказа:        " + row["articles"] + "\n\n" +
                "Статус заказа:         " + row["status"] + "\n\n" +
                "Адрес пункта выдачи:   " + row["point"];

            Label date = new Label();
            date.Bounds = new Rectangle(680, 25, 220, 80);
            date.Font = new Font("Times New Roman", 12, FontStyle.Bold);
            date.Text = "Дата доставки\n\n" + Convert.ToDateTime(row["delivery_date"]).ToShortDateString();

            card.Controls.Add(info);
            card.Controls.Add(date);

            if (userRole.Trim() == "Администратор")
            {
                // Редактирование по нажатию на карточку
                card.Click += (s, e) => EditOrder(id);
                info.Click += (s, e) => EditOrder(id);
                date.Click += (s, e) => EditOrder(id);

                // Кнопка удаления
                Button del = new Button();
                del.Text = "Удалить";
                del.Bounds = new Rectangle(20, 165, 120, 30);
                del.BackColor = Color.FromArgb(0, 250, 154);
                del.Click += (s, e) => DeleteOrder(id);

                card.Controls.Add(del);
            }

            flowOrders.Controls.Add(card);
        }

        void EditOrder(int id)
        {
            new OrderEditForm(id).ShowDialog();
            LoadOrders();
        }

        void DeleteOrder(int id)
        {
            if (MessageBox.Show("Удалить заказ?", "Подтверждение", MessageBoxButtons.YesNo) != DialogResult.Yes)
                return;

            using (var con = new NpgsqlConnection(cs))
            {
                con.Open();

                var cmd1 = new NpgsqlCommand("DELETE FROM product_in_order WHERE order_id=@id", con);
                cmd1.Parameters.AddWithValue("@id", id);
                cmd1.ExecuteNonQuery();

                var cmd2 = new NpgsqlCommand("DELETE FROM orders WHERE id=@id", con);
                cmd2.Parameters.AddWithValue("@id", id);
                cmd2.ExecuteNonQuery();
            }

            LoadOrders();
        }

        private void buttonAddOrder_Click(object sender, EventArgs e)
        {
            new OrderEditForm().ShowDialog();
            LoadOrders();
        }

        private void buttonBack_Click(object sender, EventArgs e)
        {
            Close();
        }
    }
}
